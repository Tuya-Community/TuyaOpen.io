еҝ«йҖҹејҖе§ӢвҖӢ

  1. зЎ®дҝқдҪ е·Іе®ҢжҲҗ TuyaOpen еҹәзЎҖзҺҜеўғжҗӯе»әпјҢе№¶иҝӣе…Ҙ TuyaOpen д»“еә“ж №зӣ®еҪ•гҖӮ

  1. жҢүйңҖејҖеҗҜеӨ–и®ҫејҖе…іпјҡиҝӣе…Ҙ Choice a board вҶ’ LINUX вҶ’ TKL Board ConfigurationпјҢеӢҫйҖү ENABLE_GPIO/ENABLE_I2C/ENABLE_SPI/ENABLE_PWM/ENABLE_UARTгҖӮ

  1. иҝӣе…ҘеҜ№еә”зӨәдҫӢзӣ®еҪ•пјҲеҰӮ examples/peripherals/gpioпјүпјҢжү§иЎҢ tos.py build зј–иҜ‘пјҢз„¶еҗҺз”Ё sudo иҝҗиЎҢз”ҹжҲҗзҡ„ *.elfгҖӮ

иҜҙжҳҺпјҲзј–иҜ‘ж–№ејҸпјүпјҡRaspberry Pi ж”ҜжҢҒдәӨеҸүзј–иҜ‘дёҺжң¬ең°зј–иҜ‘дёӨз§Қж–№ејҸпјӣжһ„е»әж—¶дјҡж №жҚ®еҪ“еүҚе№іеҸ°иҮӘеҠЁйҖүжӢ©еҗҲйҖӮзҡ„зј–иҜ‘ж–№ејҸгҖӮ

йҖҡз”ЁиҜҙжҳҺвҖӢ

  • жқғйҷҗпјҡеӨ–и®ҫзӨәдҫӢйҖҡеёёйңҖиҰҒи®ҝй—® /dev/* жҲ– /sys/class/*пјҢе»әи®®еңЁж ‘иҺ“жҙҫдёҠиҝҗиЎҢж—¶дҪҝз”Ё sudoгҖӮ

  • TUYA_GPIO_IRQ_LOW/HIGH еұһдәҺвҖңиҝ‘дјје®һзҺ°вҖқпјҡеә•еұӮдҪҝз”Ёиҫ№жІҝдәӢд»¶зӣ‘еҗ¬пјҢеҶҚиҜ»еҸ–еҪ“еүҚз”өе№іеҒҡиҝҮж»ӨпјҢдёҚзӯүеҗҢзЎ¬д»¶з”өе№іи§ҰеҸ‘дёӯж–ӯгҖӮ

// #define EXAMPLE_OUTPUT_PIN ...
// #define EXAMPLE_INPUT_PIN  ...

static void gpio_min_demo(void)
{
  TUYA_GPIO_BASE_CFG_T out_cfg = {
    .mode   = TUYA_GPIO_PUSH_PULL,
    .direct = TUYA_GPIO_OUTPUT,
    .level  = TUYA_GPIO_LEVEL_LOW,
  };
  TUYA_GPIO_BASE_CFG_T in_cfg = {
    .mode   = TUYA_GPIO_PULLUP,
    .direct = TUYA_GPIO_INPUT,
  };

  tkl_gpio_init(EXAMPLE_OUTPUT_PIN, &out_cfg);
  tkl_gpio_init(EXAMPLE_INPUT_PIN,  &in_cfg);

  while (1) {
    static uint8_t level = 0;
    TUYA_GPIO_LEVEL_E in_level = TUYA_GPIO_LEVEL_LOW;

    level ^= 1;
    tkl_gpio_write(EXAMPLE_OUTPUT_PIN, level ? TUYA_GPIO_LEVEL_HIGH : TUYA_GPIO_LEVEL_LOW);

    tkl_gpio_read(EXAMPLE_INPUT_PIN, &in_level);
    PR_NOTICE("GPIO in=%d out=%d", (int)in_level, (int)level);

    tal_system_sleep(1000);
  }
}

  • еҜ„еӯҳеҷЁиҜ»еҸ–зҡ„еёёи§Ғз»„еҗҲдәӢеҠЎпјҲRepeated Startпјү

  • ең°еқҖжҺўжөӢпјҲжү«жҸҸпјү

  • д»ҺжңәжЁЎејҸпјҲSlaveпјүпјҡtkl_i2c_set_slave_addr()гҖҒtkl_i2c_slave_send()гҖҒtkl_i2c_slave_receive()гҖӮ
  • дёӯж–ӯ/дәӢд»¶еӣһи°ғпјҡtkl_i2c_irq_init()гҖҒtkl_i2c_irq_enable()гҖҒtkl_i2c_irq_disable()гҖӮ
  • жү©еұ•жҺ§еҲ¶/зҠ¶жҖҒжҹҘиҜўпјҡtkl_i2c_ioctl()гҖҒtkl_i2c_get_status()гҖӮ

еҸӮиҖғвҖӢ

  • PWM зҡ„жҺҘеҸЈе®ҡд№үгҖҒеҸӮж•°иҜҙжҳҺдёҺйҖӮй…ҚжіЁж„ҸдәӢйЎ№иҜ·еҸӮиҖғ PWMй©ұеҠЁгҖӮ

и®ҫеӨҮиҠӮзӮ№жҳ е°„пјҲдёҺ FAKE дёІеҸЈејҖе…ізӣёе…іпјүвҖӢ

  • еҪ“ TKL_UART_REDIRECT_LOG_TO_STDOUT = nпјҲе…ій—ӯдёІеҸЈйҮҚе®ҡеҗ‘пјҢдҪҝз”Ёзңҹе®һзЎ¬д»¶ UARTпјүж—¶пјҢй»ҳи®Өжҳ е°„дёәпјҡ

дёІеҸЈйҮҚе®ҡеҗ‘пјҲDummy дёІеҸЈпјҡstdin/stdout/UDPпјүиҜҙжҳҺвҖӢ

еҪ“ејҖеҗҜйҮҚе®ҡеҗ‘пјҲDummy дёІеҸЈпјүеҗҺпјҢtkl_uart.c зҡ„иЎҢдёәеӨ§иҮҙеҰӮдёӢпјҲдёҺзңҹе®һ UART дёҚеҗҢпјҢдё»иҰҒз”ЁдәҺиҒ”и°ғ/жј”зӨәпјүпјҡ

#include <string.h>

#include "tal_api.h"
#include "tkl_uart.h"

static OPERATE_RET uart_loopback_test(TUYA_UART_NUM_E port)
{
  TUYA_UART_BASE_CFG_T cfg = {0};
  cfg.baudrate = 115200;
  cfg.databits = TUYA_UART_DATA_LEN_8BIT;
  cfg.parity   = TUYA_UART_PARITY_TYPE_NONE;
  cfg.stopbits = TUYA_UART_STOP_LEN_1BIT;
  cfg.flowctrl = TUYA_UART_FLOWCTRL_NONE;

  OPERATE_RET ret = tkl_uart_init(port, &cfg);
  if (ret != OPRT_OK) {
    return ret;
  }

  const uint32_t timeout_ms = 5000;
  const int bufsize = 8;
  uint8_t tx[bufsize];
  uint8_t rx[bufsize];

  for (int i = 0; i < bufsize; i++) {
    tx[i] = (uint8_t)('A' + i);
  }

  for (int round = 0; round < 3; round++) {
    memset(rx, 0, sizeof(rx));

    int wr = tkl_uart_write(port, tx, sizeof(tx));
    if (wr != (int)sizeof(tx)) {
      ret = OPRT_COM_ERROR;
      break;
    }

    int got = 0;
    SYS_TIME_T start = tal_system_get_millisecond();
    while (got < (int)sizeof(rx)) {
      SYS_TIME_T now = tal_system_get_millisecond();
      if ((uint32_t)(now - start) > timeout_ms) {
        ret = OPRT_TIMEOUT;
        break;
      }

      int rd = tkl_uart_read(port, rx + got, (uint32_t)sizeof(rx) - (uint32_t)got);
      if (rd > 0) {
        got += rd;
      } else {
        tal_system_sleep(5);
      }
    }

    if (ret != OPRT_OK) {
      break;
    }
    if (memcmp(tx, rx, sizeof(tx)) != 0) {
      ret = OPRT_COM_ERROR;
      break;
    }
  }

  tkl_uart_deinit(port);
  return ret;
}

  • е»әи®®з”Ё sudo иҝҗиЎҢпјҢжҲ–зЎ®дҝқеҪ“еүҚз”ЁжҲ·еұһдәҺ audio з»„пјҲеҗҰеҲҷеҸҜиғҪж— жі•жү“ејҖеЈ°еҚЎи®ҫеӨҮиҠӮзӮ№пјүгҖӮ
